## openwrt配置  
```
{  
  "reverse":{
    "bridges":[  
      {  
        "tag":"bridge",
        "domain":"web.hhxx.bf"
      }
    ]
  },
  "outbounds":[
    {
      "tag":"tunnel",
      "protocol":"vmess",
      "settings":{  
        "vnext":[  
          {  
            "address":"vps的ip",
            "port":openwrt配置的端口,
            "users":[  
              {  
                "id":"openwrt配置的id",
                "alterId":0
              }
            ]
          }
        ]
      }
    },
    {  
      "protocol":"freedom",
      "settings":{  
      },
      "tag":"out"
    }
  ],
  "routing":{  
    "rules":[  
      {
        "type":"field",
        "inboundTag":[  
          "bridge"
        ],
        "domain":[  
          "full:web.hhxx.bf"
        ],
        "outboundTag":"tunnel"
      },
      {
        "type":"field",
        "inboundTag":[  
          "bridge"
        ],
        "outboundTag":"out"
      }
    ]    
  }
}
```  

## vps配置  
```
{  
  "reverse":{
    "portals":[  
      {  
        "tag":"portal",
        "domain":"web.hhxx.bf"
      }
    ]
  },
  "inbounds":[
    {
      "tag":"tunnel",
      "port":vps使用的端口,
      "protocol":"vmess",
      "settings":{  
        "clients":[  
          {  
            "id":"vps使用的id",
            "alterId":0
          }
        ]
      }
    }, 
    {  
      "tag": "interconn",//
      "port":openwrt使用的端口,
      "protocol":"vmess",
      "settings":{  
        "clients":[  
          {  
            "id":"openwrt使用的id",
            "alterId":0
          }
        ]
      }
    }
  ],
  "routing":{   
    "rules":[  
      {
        "type":"field",
        "inboundTag":[  
          "external"
        ],
        "outboundTag":"portal"
      },
      {
        "type":"field",
        "inboundTag":[  
          "tunnel"
        ],
        "domain":[  
          "full:web.hhxx.bf"
        ],
        "outboundTag":"portal"
      }
    ]
  }
}
```
