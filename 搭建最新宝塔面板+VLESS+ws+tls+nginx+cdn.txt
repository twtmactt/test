1.域名解析，安装bbr

2.安装v2ray
   见v2flay安装v2ray

3.安装BT面板
  一切都安装完后，点击「网站」——「添加站点」(默认80端口)
  成功添加完站点后，点击「设置」——SSL——Let’s Encrypt，成功申请SSL后，保存
  再点击站点「设置」的「配置文件」选项，在ssl最后一个}前添加如下代码：
  location /ws
{
    proxy_pass http://127.0.0.1:你的端口号;
    proxy_redirect off;
    proxy_http_version 1.1;
    proxy_set_header Upgrade $http_upgrade;
    proxy_set_header Connection "upgrade";
    proxy_set_header Host $http_host;
    proxy_read_timeout 300s;
}
	回到「首页」，重启「Nignx服务」
	BT面板「安全」，防火墙端口放行，新增你的V2Ray端口

4.修改V2Ray配置文件(vi /etc/v2ray/config.json)
   {
  "log": {
        "access": "/var/log/v2ray/access.log",
        "error": "/var/log/v2ray/error.log",
        "loglevel": "warning"
    },
  "inbounds": [
    {
    "port":端口,
      "listen": "ip",
      "tag": "VLESS-in",
      "protocol": "VLESS",
      "settings": {
        "clients": [
          {
          "id":"UUID",
            "alterId": 0
          }
        ],
            "decryption": "none"
      },
      "streamSettings": {
        "network": "ws",
        "wsSettings": {
          "path":"/ws" 
        }
      }
    }
  ],
  "outbounds": [
    {
      "protocol": "freedom",
      "settings": { },
      "tag": "direct"
    },
    {
      "protocol": "blackhole",
      "settings": { },
      "tag": "blocked"
    }
  ],
  "dns": {
    "servers": [
      "https+local://1.1.1.1/dns-query",
          "1.1.1.1",
          "1.0.0.1",
          "8.8.8.8",
          "8.8.4.4",
          "localhost"
    ]
  },
  "routing": {
    "domainStrategy": "AsIs",
    "rules": [
      {
        "type": "field",
        "inboundTag": [
          "VLESS-in"
        ],
        "outboundTag": "direct"
      }
    ]
  }
}

5.重启V2Ray服务
  service v2ray restart
  service v2ray status #通过该命令，可查看v2ray是否运行成功
